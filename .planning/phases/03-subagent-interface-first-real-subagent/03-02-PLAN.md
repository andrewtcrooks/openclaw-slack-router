1. Create `src/gateway/chat-client.ts` implementing the `gatewayChatSend` function. Follow the research Pattern 3 implementation closely (03-RESEARCH.md lines 197-348). The function:

   - Exports: `GatewayChatOptions` interface with `{ url: string; token?: string; timeoutMs?: number }`, `ChatSendResult` interface with `{ text: string; runId: string }`, and `gatewayChatSend` async function.
   - Opens a WebSocket to `options.url` using `import WebSocket from "ws";`
   - Uses `import { randomUUID } from "node:crypto";` for request IDs
   - Implements the openclaw gateway protocol:
     a. Waits for `connect.challenge` event, extracts `nonce`
     b. Sends `connect` request with: `{ minProtocol: 1, maxProtocol: 1, client: { id: "slack-router", version: "0.1.0", platform: process.platform, mode: "backend" }, caps: [], role: "operator", scopes: ["operator.admin"] }` plus `auth: { token }` if token provided
     c. On connect response `ok: true`, sends `chat.send` with `{ sessionKey, message, idempotencyKey }`
     d. On `chat.send` ack, stores `runId` from `payload.runId`
     e. Listens for `chat` events matching `runId`:
        - `state: "delta"`: accumulates text in `deltaText` variable (delta text is cumulative)
        - `state: "final"`: extracts text via `extractTextFromMessage()`, resolves promise
        - `state: "error"`: rejects with `payload.errorMessage`
        - `state: "aborted"`: rejects with "gateway chat aborted"
   - Has a configurable timeout (default 120_000ms) that rejects the promise and closes the WebSocket
   - Handles WebSocket `error` and `close` events — rejects if not already settled
   - Internal `extractTextFromMessage(message: unknown): string` helper:
     - If `message.content` is a string, return it
     - If `message.content` is an array, filter for `type === "text"` blocks, join `.text` values with `\n`
     - If `message.text` is a string, return it
     - Otherwise return empty string

1. Create `src/subagents/openclaw-gateway.ts` per D-02, D-10, D-11, D-15:

```typescript
import type { SubagentDefinition } from "./types.js";
import type { SubagentContext } from "../types.js";
import { gatewayChatSend } from "../gateway/chat-client.js";
import type { GatewayChatOptions } from "../gateway/chat-client.js";
import { randomUUID } from "node:crypto";

export interface OpenclawGatewaySubagentOptions {
  gatewayUrl: string;    // ws://127.0.0.1:18789
  gatewayToken?: string; // optional OPENCLAW_GATEWAY_TOKEN
  timeoutMs?: number;    // default 120_000
}

export function createOpenclawGatewaySubagent(
  options: OpenclawGatewaySubagentOptions,
): SubagentDefinition {
  const { gatewayUrl, gatewayToken, timeoutMs } = options;

  return {
    name: "openclaw-gateway",
    description: "Routes messages through the local openclaw gateway",
    async handle(ctx: SubagentContext): Promise<string> {
      const chatOptions: GatewayChatOptions = {
        url: gatewayUrl,
        token: gatewayToken,
        timeoutMs,
      };

      // D-11: sessionKey = ctx.channelId (e.g., "C08ABC123")
      // D-15: idempotencyKey = UUID per dispatch
      const idempotencyKey = randomUUID();

      try {
        const result = await gatewayChatSend(chatOptions, {
          sessionKey: ctx.channelId,
          message: ctx.currentMessage,
          idempotencyKey,
        });
        return result.text;
      } catch (err) {
        // Discretion: catch gateway failures, return error string
        const errorMsg = err instanceof Error ? err.message : String(err);
        return `[openclaw-gateway error] ${errorMsg}`;
      }
    },
  };
}
```
